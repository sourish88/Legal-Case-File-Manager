# Logging Configuration Examples for Legal Case File Manager

# Development Environment (.env for development)
LOG_LEVEL=DEBUG
LOG_FORMAT=development
LOG_FILE=logs/app.log
LOG_MAX_BYTES=10485760
LOG_BACKUP_COUNT=5
LOG_ENABLE_CONSOLE=true

# Production Environment (.env for production)
LOG_LEVEL=WARNING
LOG_FORMAT=json
LOG_FILE=logs/app.log
LOG_MAX_BYTES=52428800
LOG_BACKUP_COUNT=10
LOG_ENABLE_CONSOLE=false

# Testing Environment (.env for testing)
LOG_LEVEL=ERROR
LOG_FORMAT=development
LOG_FILE=logs/test.log
LOG_MAX_BYTES=5242880
LOG_BACKUP_COUNT=3
LOG_ENABLE_CONSOLE=true

# Log Levels Available:
# - DEBUG: Detailed information, typically of interest only when diagnosing problems
# - INFO: Confirmation that things are working as expected
# - WARNING: An indication that something unexpected happened
# - ERROR: Due to a more serious problem, the software has not been able to perform some function
# - CRITICAL: A serious error, indicating that the program itself may be unable to continue running

# Log Formats Available:
# - development: Human-readable format with colors (good for development)
# - json: Structured JSON format (good for production and log aggregation)

# File Rotation:
# - LOG_MAX_BYTES: Maximum size of each log file before rotation (in bytes)
# - LOG_BACKUP_COUNT: Number of backup files to keep
